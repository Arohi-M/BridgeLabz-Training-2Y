public class KmToMiles {
    public static void main(String[] args) {

        // given distance in km
        double km = 10.8;

        // 1 km = 1.6 miles
        double miles = km * 1.6;

        // output
        System.out.println("The distance " + km + " km in miles is " + miles);
    }
}
